I. Precise Definitions (Non-Overlapping, Enforceable)
-----------------------------------------------------

### 1\. **Summary Description**

**Definition:**  
A concise, factual encapsulation of _what exists or will exist_, stripped of rationale, intent, or justification.

**Function:**  
Orientation and scope anchoring.

**Test:**  
If removed, stakeholders lose immediate context.

* * *

### 2\. **Purpose**

**Definition:**  
The fundamental _reason for existence_ — the enduring “why” that does **not change** with tactics or time.

**Function:**  
North star for decision-making and trust alignment.

**Test:**  
If the activity succeeded perfectly, _why would it matter_?

* * *

### 3\. **Goal**

**Definition:**  
A **specific future state** that satisfies the purpose within a defined time horizon.

**Function:**  
Directional commitment.

**Test:**  
Can be clearly true or false at a point in time?

* * *

### 4\. **Objectives**

**Definition:**  
Discrete, measurable **conditions that must be met** to achieve the goal.

**Function:**  
Operational decomposition of the goal.

**Test:**  
Failure of any objective materially jeopardizes the goal.

* * *

### 5\. **Results**

**Definition:**  
Direct, observable **outputs** produced by executed actions.

**Function:**  
Evidence of execution.

**Test:**  
Could an auditor verify this without interpretation?

* * *

### 6\. **Outcomes**

**Definition:**  
Causal **effects created by the results** within the target system or population.

**Function:**  
Value realization.

**Test:**  
Would this have occurred without the results?

* * *

### 7\. **Performance Expectations**

**Definition:**  
Pre-declared **standards of quality, speed, scale, and reliability** applied to actions and results.

**Function:**  
Trust calibration and accountability.

**Test:**  
Can performance be graded objectively against this?

* * *

### 8\. **Qualifications**

**Definition:**  
Explicit **constraints, assumptions, dependencies, and exclusions** governing interpretation.

**Function:**  
Risk containment and misalignment prevention.

**Test:**  
Without this, could results be misrepresented?

Operating principle: “Rule of 7s” as an intent-extraction engine
----------------------------------------------------------------

Use **7 stages**, each with **≤7 questions**, each question with **≤7 answer options** (or constrained free-text). This matches how people reliably chunk, compare, and commit decisions without cognitive overload, while still producing **granular, machine-usable intent**.

Output: a structured **Idea Injection → Project → Actions** payload that can be:

*   rendered as a GUI,
*   converted into engineered AI prompts,
*   stored to Firebase (Firestore),
*   and audited later for Proof-of-Action → Influence → Trust → Outcome.

```
/**
 * Deterministic prompt assembly logic (versioned).
 * - No LLM calls
 * - Stable ordering
 * - Stable placeholder substitution
 * - Stable hashing for cache + change detection
 */

type IntentPayload = Record<string, any>;

type StagePromptTemplate = {
  promptBlockId: string;
  stageId: string;
  templateVersion: string;
  template: string;
  requiredFields?: string[];
};

type MasterPromptTemplate = {
  templateVersion: string;
  stagePromptBlockOrder: string[]; // ["PB_S1"...]
  template: string; // includes {{PB_S1}} tokens
};

type Engine = {
  engineVersion: string;
  promptTemplates: { stage: StagePromptTemplate[]; master: MasterPromptTemplate };
};

function getValueByPath(obj: any, path: string): any {
  // path like "identity.title"
  return path.split(".").reduce((acc, k) => (acc == null ? undefined : acc[k]), obj);
}

function normalizeValue(v: any): string {
  if (v === undefined || v === null) return "";
  if (Array.isArray(v)) return v.join(", ");
  if (typeof v === "object") return JSON.stringify(v);
  return String(v);
}

/**
 * Deterministic substitution:
 * - Replace {{field}} with normalizedValue
 * - Unresolved placeholders become empty string (optional: throw in strict mode)
 */
function renderTemplate(template: string, payload: IntentPayload, strict = false): string {
  return template.replace(/\{\{([^}]+)\}\}/g, (_, tokenRaw) => {
    const token = tokenRaw.trim();
    // Allow PB_ tokens to pass through here; they get rendered at master stage
    if (token.startsWith("PB_") || token === "engineVersion") {
      if (token === "engineVersion") return normalizeValue(payload.engineVersion);
      return `{{${token}}}`;
    }
    const v = getValueByPath(payload, token);
    if ((v === undefined || v === null || v === "") && strict) {
      throw new Error(`Missing required field for template token: ${token}`);
    }
    return normalizeValue(v);
  });
}

function sortStageTemplatesDeterministically(templates: StagePromptTemplate[]): StagePromptTemplate[] {
  return [...templates].sort((a, b) => a.promptBlockId.localeCompare(b.promptBlockId));
}

function buildStageBlocks(engine: Engine, intent: IntentPayload, strict = true): Record<string, string> {
  const blocks: Record<string, string> = {};
  const templates = sortStageTemplatesDeterministically(engine.promptTemplates.stage);

  for (const t of templates) {
    // optional requiredFields check (strict governance)
    if (strict && t.requiredFields?.length) {
      for (const f of t.requiredFields) {
        const v = getValueByPath(intent, f);
        if (v === undefined || v === null || v === "") {
          throw new Error(`Missing required field '${f}' for promptBlockId ${t.promptBlockId}`);
        }
      }
    }
    blocks[t.promptBlockId] = renderTemplate(t.template, { ...intent, engineVersion: engine.engineVersion }, strict);
  }
  return blocks;
}

/**
 * Assemble master prompt deterministically:
 * - Render master template with engineVersion
 * - Inject stage blocks in defined order
 */
function buildMasterPrompt(engine: Engine, intent: IntentPayload, strict = true): string {
  const stageBlocks = buildStageBlocks(engine, intent, strict);

  // First render engineVersion in master template
  let master = engine.promptTemplates.master.template.replace(/\{\{engineVersion\}\}/g, engine.engineVersion);

  // Inject blocks in declared order (enforces stability)
  for (const pb of engine.promptTemplates.master.stagePromptBlockOrder) {
    const block = stageBlocks[pb] ?? "";
    master = master.replace(new RegExp(`\\{\\{${pb}\\}\\}`, "g"), block);
  }

  // Any remaining PB tokens are blanked (deterministic)
  master = master.replace(/\{\{PB_S[1-7]\}\}/g, "");

  return master;
}

/**
 * Deterministic hash input:
 * - stable JSON stringify (keys sorted)
 * - sha256
 * NOTE: Provide your own stable stringify + crypto implementation for your runtime.
 */
function stableStringify(obj: any): string {
  if (obj === null || typeof obj !== "object") return JSON.stringify(obj);
  if (Array.isArray(obj)) return `[${obj.map(stableStringify).join(",")}]`;
  const keys = Object.keys(obj).sort();
  return `{${keys.map(k => JSON.stringify(k) + ":" + stableStringify(obj[k])).join(",")}}`;
}
```

5) Deterministic, versioned prompt assembly logic (now “prompt pack” per function)
==================================================================================

Two-layer approach:

### Layer 1 — Deterministic plan generation (rules-based)

*   Input: intent payload (A–E)
*   Output: plan object (objectives, requirements, tasks, evidencePlan)
*   Determinism: **no LLM** in generation step (or if LLM is used, you must store the model + temperature + seed + raw output and treat it as non-deterministic; recommended: keep generation rules-based).

### Layer 2 — Prompt pack generation (template substitution)

*   Input: intent payload + generated plan
*   Output: prompt snippets per AI function
*   Determinism: stable ordering + stable template versions + stable placeholder substitution + stable hashing

```
type Intent = Record<string, any>;
type Plan = Record<string, any>;

type PromptTemplate = {
  promptId: string;                 // e.g., "P_ENGINEERING_V1"
  templateVersion: string;          // semver
  aiFunction: "planner" | "ux" | "engineering" | "data" | "legal" | "gtm" | "qa" | "pmo";
  template: string;                 // placeholders: {{intent.summary.title}}, {{plan.tasks}}
  requiredPaths?: string[];         // strict-mode gating
};

type EngineConfig = {
  engineVersion: string;
  promptTemplates: PromptTemplate[];
  promptPackOrder: string[];        // stable order of promptIds
};

function get(obj: any, path: string): any {
  return path.split(".").reduce((acc, k) => (acc == null ? undefined : acc[k]), obj);
}

function normalize(v: any): string {
  if (v === undefined || v === null) return "";
  if (Array.isArray(v)) return v.map(x => (typeof x === "string" ? x : JSON.stringify(x))).join("\n");
  if (typeof v === "object") return JSON.stringify(v, null, 2);
  return String(v);
}

function render(template: string, ctx: any, strict = true): string {
  return template.replace(/\{\{([^}]+)\}\}/g, (_, raw) => {
    const token = raw.trim();
    const val = get(ctx, token);
    if (strict && (val === undefined || val === null || val === "")) {
      throw new Error(`Missing token: ${token}`);
    }
    return normalize(val);
  });
}

function stableStringify(obj: any): string {
  if (obj === null || typeof obj !== "object") return JSON.stringify(obj);
  if (Array.isArray(obj)) return `[${obj.map(stableStringify).join(",")}]`;
  const keys = Object.keys(obj).sort();
  return `{${keys.map(k => JSON.stringify(k) + ":" + stableStringify(obj[k])).join(",")}}`;
}

// Replace with your runtime sha256
function pseudoHash(s: string): string {
  let h = 0;
  for (let i = 0; i < s.length; i++) h = (h * 31 + s.charCodeAt(i)) >>> 0;
  return "h32:" + h.toString(16);
}

export function buildPromptPack(cfg: EngineConfig, intent: Intent, plan: Plan, strict = true) {
  const ctx = { engineVersion: cfg.engineVersion, intent, plan };

  const byId = new Map(cfg.promptTemplates.map(t => [t.promptId, t]));
  const prompts: { promptId: string; aiFunction: string; templateVersion: string; promptText: string; inputsHash: string }[] = [];

  for (const pid of cfg.promptPackOrder) {
    const t = byId.get(pid);
    if (!t) throw new Error(`Unknown promptId in order: ${pid}`);

    if (strict && t.requiredPaths?.length) {
      for (const p of t.requiredPaths) {
        const v = get(ctx, p);
        if (v === undefined || v === null || v === "") throw new Error(`Missing required path '${p}' for ${pid}`);
      }
    }

    const promptText = render(t.template, ctx, strict);
    const inputsHash = pseudoHash(stableStringify({
      engineVersion: cfg.engineVersion,
      promptId: t.promptId,
      templateVersion: t.templateVersion,
      intent,
      plan
    }));

    prompts.push({ promptId: t.promptId, aiFunction: t.aiFunction, templateVersion: t.templateVersion, promptText, inputsHash });
  }

  return prompts;
}
```

```
type Intent = any;

type Objective = {
  id: string;
  workstream: string;
  title: string;
  acceptanceCriteria: string[];
  evidenceTypes: string[];
};

type Task = {
  id: string;
  workstream: string;
  title: string;
  dependsOn: string[];
  evidenceTypes: string[];
};

type Plan = {
  objectives: Objective[];
  requirements: { functional: string[]; nonFunctional: string[] };
  tasks: Task[];
  evidencePlan: { artifact: string; proves: string }[];
  firestoreModel: any;
};

type Rule = any;
type CategoryDefaults = any;

function stableSortById<T extends { id: string }>(arr: T[]): T[] {
  return [...arr].sort((a, b) => a.id.localeCompare(b.id));
}

function dedupeById<T extends { id: string }>(arr: T[]): T[] {
  const m = new Map<string, T>();
  for (const x of arr) m.set(x.id, x);
  return [...m.values()];
}

function applyRules(intent: Intent, base: { objectives: Objective[]; tasks: Task[]; workstreams: string[] }, rules: Rule[]) {
  const out = { ...base, objectives: [...base.objectives], tasks: [...base.tasks], workstreams: [...base.workstreams] };

  // This matcher is intentionally limited to a few operators to keep it deterministic and safe.
  const match = (when: any): boolean => {
    if (when.anyEq) {
      const v = get(intent, when.anyEq);
      return when.values.includes(v);
    }
    if (when.anyIn) {
      const arr = get(intent, when.anyIn[0]) ?? get(intent, when.anyIn); // support string or array
      if (!Array.isArray(arr)) return false;
      return arr.some((x: any) => when.values.includes(x));
    }
    if (when.lt) return Number(get(intent, when.lt)) < Number(when.value);
    if (when.gt) return Number(get(intent, when.gt)) > Number(when.value);
    return false;
  };

  for (const r of rules) {
    if (match(r.when)) {
      if (r.then.addWorkstreams) out.workstreams.push(...r.then.addWorkstreams);
      if (r.then.addObjectives) out.objectives.push(...r.then.addObjectives);
      if (r.then.addTasks) out.tasks.push(...r.then.addTasks);
    }
  }

  out.workstreams = [...new Set(out.workstreams)].sort();
  out.objectives = stableSortById(dedupeById(out.objectives));
  out.tasks = stableSortById(dedupeById(out.tasks));

  return out;
}

function get(obj: any, path: string): any {
  return path.split(".").reduce((acc, k) => (acc == null ? undefined : acc[k]), obj);
}

export function generatePlanDeterministic(
  intent: Intent,
  categoryDefaults: CategoryDefaults,
  rules: Rule[],
  firestoreModelTemplate: any
): Plan {
  const category = intent.summary?.category || intent.intent?.summary?.category || "other";
  const defaults = categoryDefaults[category] || categoryDefaults["product"];

  const baseWorkstreams = defaults.defaultWorkstreams || [];
  const baseObjectives: Objective[] = defaults.objectiveTemplates || [];
  const baseTasks: Task[] = defaults.taskTemplates || [];

  const applied = applyRules(intent, { objectives: baseObjectives, tasks: baseTasks, workstreams: baseWorkstreams }, rules);

  // Requirements: deterministic composition (category + compliance + determinism mandates)
  const functional = [
    "5-stage intent capture (A–E) with save/resume",
    "Deterministic plan generation + hashing",
    "Prompt pack generation + export",
    "Audit trail for runs/artifacts"
  ];

  const nonFunctional = [
    "Same inputs -> same outputs",
    "Versioned engine + templates",
    "Least-privilege access controls"
  ];

  const compliance = intent.qualifications?.compliance || [];
  if (Array.isArray(compliance) && compliance.length) nonFunctional.push(`Compliance alignment: ${compliance.join(", ")}`);

  const plan: Plan = {
    objectives: applied.objectives,
    tasks: applied.tasks,
    requirements: { functional, nonFunctional },
    evidencePlan: [
      { artifact: "inputsHash", proves: "Run inputs integrity" },
      { artifact: "planHash", proves: "Deterministic output integrity" }
    ],
    firestoreModel: firestoreModelTemplate
  };

  // Stable ordering already enforced above.
  return plan;
}
```
