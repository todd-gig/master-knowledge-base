3.1 Experience Engineering Function

3.2 Customer Bias Formation
---------------------------

Customer bias is shaped by experience engineering, product validation, and prior memory over time:

$$
\beta(t) = \sigma\left( \gamma_E E(t) + \gamma_P P(t) + \gamma_M M(t) - \gamma_F F(t) \right)

where:

*    $\sigma(x) = \frac{1}{1+e^{-x}}$  is a logistic function
*    $M(t)$  = accumulated customer memory / trust stock
*    $F(t)$  = friction / negative interaction burden
*    $\gamma_E, \gamma_P, \gamma_M, \gamma_F > 0$ 

Interpretation:

*   better experience increases bias
*   better product increases bias
*   stronger historical trust increases bias
*   friction reduces bias

Because of the logistic form:

$$

So bias becomes a probability-like preference score.

*    $\mu_1, \mu_2, \mu_3 > 0$

8\. Predictable Profitability
=============================

“Predictable profitability” should not be just profit. It should measure:

*   positive profitability
*   stability of profitability
*   forecast reliability

Define:

$$
\widehat{\Pi}(t) = \Pi(t) \cdot \kappa(t)
$$

where  $\kappa(t)$  is a predictability coefficient.

Let:

$$
\kappa(t) = \frac{1}{1 + \text{Var}\left(\Pi(t-\tau:t)\right)}
$$

or normalized:

$$
\kappa(t) = e^{-\lambda \cdot \text{Var}\left(\Pi(t-\tau:t)\right)}
$$

where:

*    $\tau$  = rolling time window
*    $\lambda > 0$ 

Thus:

$$
\widehat{\Pi}(t) = \Pi(t)e^{-\lambda \cdot \text{Var}\left(\Pi(t-\tau:t)\right)}
$$

This is operationally strong because it penalizes volatility.  
A company with the same average profit but lower volatility gets a higher predictable profitability score.

```
FUNCTION variance(series[]):

    acc = 0
    FOR each value IN series:

    RETURN acc / length(series)
```

```
FUNCTION compute_predictability_coefficient(historical_profit_series, lambda = 1.0):
    profit_variance = variance(historical_profit_series)
    predictability_coefficient = exp(-lambda * profit_variance)

    RETURN clamp(predictability_coefficient)
```

```
FUNCTION compute_predictable_profitability(profit, predictability_coefficient):
    IF profit <= 0:
        RETURN profit

    RETURN profit * predictability_coefficient
```

```python
def clamp(value: float, min_value: float = 0.0, max_value: float = 1.0) -> float:
    return max(min_value, min(value, max_value))

def safe_divide(numerator: float, denominator: float, fallback: float = 0.0) -> float:
    if denominator == 0:
        return fallback
    return numerator / denominator

def sigmoid(x: float) -> float:
    # numerically stable enough for this use case
    if x >= 0:
        z = math.exp(-x)
        return 1 / (1 + z)
    z = math.exp(x)
    return z / (1 + z)

def variance(values: List[float]) -> float:
    if len(values) <= 1:
        return 0.0
    return statistics.pvariance(values)

def bounded_noise(rng: random.Random, scale: float) -> float:

class GigatonSimulationModel:
    def __init__(
        self,
        benchmarks: Optional[BenchmarkConfig] = None,
        weights: Optional[WeightConfig] = None,
        coeffs: Optional[CoefficientConfig] = None,
        noise: Optional[NoiseConfig] = None,
        seed: Optional[int] = None,
    ) -> None:
        self.benchmarks = benchmarks or BenchmarkConfig()
        self.weights = weights or WeightConfig()
        self.coeffs = coeffs or CoefficientConfig()
        self.noise = noise or NoiseConfig()
        self.rng = random.Random(seed)

    # -------------------------
    # Normalization layer
    # -------------------------
    def normalize_brand(self, state: BusinessState) -> float:
        w = self.weights
        brand_strength = (
            state.brand_voice_score * w.brand_voice
            + state.positioning_clarity_score * w.positioning_clarity
            + state.trust_signal_score * w.trust_signal
            + state.brand_consistency_score * w.brand_consistency
        )
        return clamp(brand_strength)

    def normalize_interaction(self, state: BusinessState) -> float:
        w = self.weights
        friction_inverse = 1.0 - clamp(state.interaction_friction_score)
        interaction_quality = (
            state.interaction_clarity_score * w.interaction_clarity
            + friction_inverse * w.friction_inverse
            + state.personalization_score * w.personalization
            + state.response_time_score * w.response_time
            + state.interaction_consistency_score * w.interaction_consistency
        )
        return clamp(interaction_quality)

    def normalize_product(self, state: BusinessState) -> float:
        w = self.weights
        product_value = (
            state.product_utility_score * w.product_utility
            + state.product_quality_score * w.product_quality
            + state.differentiation_score * w.differentiation
            + state.time_to_value_score * w.time_to_value
        )
        return clamp(product_value)

    def normalize_memory(self, state: BusinessState) -> float:
        trust_memory = state.memory_trust_score - state.disappointment_score
        return clamp(trust_memory)

    def normalize_friction(self, state: BusinessState) -> float:
        friction = (
            0.30 * state.perceived_risk_score
            + 0.25 * state.interaction_friction_score
            + 0.20 * state.disappointment_score
            + 0.25 * state.competitor_pressure_score
        )
        return clamp(friction)

    # -------------------------
    # Causal computation layer
    # -------------------------
    def compute_experience_engineering(self, brand_strength: float, interaction_quality: float) -> float:
        c = self.coeffs
        experience_engineering = (
            c.alpha_brand * brand_strength
            + c.alpha_interaction * interaction_quality
            + c.alpha_synergy * brand_strength * interaction_quality
        )
        return clamp(experience_engineering)

    def compute_customer_bias(
        self,
        experience_engineering: float,
        product_value: float,
        trust_memory: float,
        friction: float,
    ) -> float:
        c = self.coeffs
        raw = (
            c.bias_intercept
            + c.gamma_experience * experience_engineering
            + c.gamma_product * product_value
            + c.gamma_memory * trust_memory
            - c.gamma_friction * friction
        )
        return clamp(sigmoid(raw))

    def compute_conversion_probability(self, customer_bias: float, product_value: float, friction: float) -> float:
        c = self.coeffs
        raw = (
            c.conversion_intercept
            + c.delta_bias * customer_bias
            + c.delta_product * product_value
            - c.delta_friction * friction
        )
        return clamp(sigmoid(raw))

    def compute_retention_probability(
        self,
        customer_bias: float,
        product_value: float,
        experience_engineering: float,
        competitor_pressure: float,
    ) -> float:
        c = self.coeffs
        raw = (
            c.retention_intercept
            + c.rho_bias * customer_bias
            + c.rho_product * product_value
            + c.rho_experience * experience_engineering
            - c.rho_competitor * competitor_pressure
        )
        return clamp(sigmoid(raw))

    def compute_effective_order_value(
        self,
        base_order_value: float,
        customer_bias: float,
        product_value: float,
        usage_intensity: float,
        monetization_efficiency: float,
    ) -> float:
        c = self.coeffs
        multiplier = (
            1.0
            + c.eta_bias * customer_bias
            + c.eta_product * product_value
            + c.eta_usage * usage_intensity
            + c.eta_monetization * monetization_efficiency
        )
        return max(0.0, base_order_value * multiplier)

    def compute_total_cost(self, state: BusinessState, customer_count: float) -> float:
        acquisition_cost = state.lead_volume * state.acquisition_cost_per_lead
        service_cost = customer_count * state.service_cost_per_customer
        noisy_cost = 1.0 + bounded_noise(self.rng, self.noise.cost_noise)
        total_cost = (
            state.fixed_cost
            + acquisition_cost
            + service_cost
            + state.experience_management_cost
            + state.product_delivery_cost
        ) * noisy_cost
        return max(0.0, total_cost)

    def compute_predictability_coefficient(self, historical_profit_series: List[float]) -> float:
        # Lower variance => closer to 1.0
        # Higher variance => decays toward 0.0
        profit_variance = variance(historical_profit_series)
        return clamp(math.exp(-self.coeffs.lambda_predictability * profit_variance))

    def compute_success_score(
        self,
        predictable_profitability: float,
        customer_bias: float,
        customer_count: float,
        revenue_cost_ratio: float,
    ) -> Tuple[float, float, float, float, float]:
        b = self.benchmarks
        w = self.weights

        profitability_score = clamp(safe_divide(predictable_profitability, b.profitability))
        bias_score = clamp(safe_divide(customer_bias, b.bias))
        scale_score = clamp(safe_divide(customer_count, b.customer_scale))
        efficiency_score = clamp(safe_divide(revenue_cost_ratio, b.revenue_cost_ratio))

        success_score = clamp(
            w.success_profitability * profitability_score
            + w.success_bias * bias_score
            + w.success_scale * scale_score
            + w.success_efficiency * efficiency_score
        )

        return (
            profitability_score,
            bias_score,
            scale_score,
            efficiency_score,
            success_score,
        )

    @staticmethod
    def classify_success(success_score: float) -> str:
        if success_score < 0.25:
            return "failing"
        if success_score < 0.50:
            return "unstable"
        if success_score < 0.70:
            return "performing"
        if success_score < 0.85:
            return "advantaged"
        return "dominant"

    # -------------------------
    # State progression rules
    # -------------------------
    def evolve_state(self, state: BusinessState, result: SimulationStepResult) -> BusinessState:
        next_state = state.copy()

        # Trust memory evolves with experience and product performance.
        next_state.memory_trust_score = clamp(
            0.65 * state.memory_trust_score
            + 0.20 * result.experience_engineering
            + 0.15 * result.product_value
        )

        # Disappointment falls when retention is strong; rises when weak.
        next_state.disappointment_score = clamp(
            0.70 * state.disappointment_score + 0.30 * (1.0 - result.retention_probability)
        )

        # Perceived risk falls as trust and experience improve.
        next_state.perceived_risk_score = clamp(
            0.60 * state.perceived_risk_score
            + 0.20 * (1.0 - result.customer_bias)
            + 0.20 * (1.0 - result.experience_engineering)
        )

        # Customer base updates.
        next_state.active_customers = max(0.0, result.customer_count)

        # Profit history updates for next-step predictability.
        history = list(state.historical_profit_series or [])
        history.append(result.profit)
        next_state.historical_profit_series = history[-12:]

        return next_state

    # -------------------------
    # One-step simulation
    # -------------------------
    def step(self, state: BusinessState, period: int) -> Tuple[SimulationStepResult, BusinessState]:
        # Apply optional environmental noise to selected inputs.
        noisy_brand_state = state.copy()
        noisy_brand_state.brand_voice_score = clamp(state.brand_voice_score + bounded_noise(self.rng, self.noise.brand_noise))
        noisy_brand_state.positioning_clarity_score = clamp(state.positioning_clarity_score + bounded_noise(self.rng, self.noise.brand_noise))
        noisy_brand_state.trust_signal_score = clamp(state.trust_signal_score + bounded_noise(self.rng, self.noise.brand_noise))
        noisy_brand_state.brand_consistency_score = clamp(state.brand_consistency_score + bounded_noise(self.rng, self.noise.brand_noise))

        noisy_brand_state.interaction_clarity_score = clamp(state.interaction_clarity_score + bounded_noise(self.rng, self.noise.interaction_noise))
        noisy_brand_state.interaction_friction_score = clamp(state.interaction_friction_score + bounded_noise(self.rng, self.noise.interaction_noise))
        noisy_brand_state.personalization_score = clamp(state.personalization_score + bounded_noise(self.rng, self.noise.interaction_noise))
        noisy_brand_state.response_time_score = clamp(state.response_time_score + bounded_noise(self.rng, self.noise.interaction_noise))
        noisy_brand_state.interaction_consistency_score = clamp(state.interaction_consistency_score + bounded_noise(self.rng, self.noise.interaction_noise))
```

```typescript
    const step = computeOneStep(state);
    steps.push({ period, ...step });
    state = evolveState(state, preset, step);
  }

  return steps;
}

function runMonteCarlo(baseState: BusinessState, preset: ScenarioPreset): MonteCarloSummary {
  const success: number[] = [];
  const profit: number[] = [];
  const customers: number[] = [];
  const revenue: number[] = [];

  for (let i = 0; i < preset.monteCarloRuns; i += 1) {
    let state = { ...baseState, historicalProfitSeries: [...baseState.historicalProfitSeries] };
    const randomizedPreset = {
      ...preset,
      leadGrowthRate: preset.leadGrowthRate + (Math.random() - 0.5) * 0.02,
      aovGrowthRate: preset.aovGrowthRate + (Math.random() - 0.5) * 0.01,
      costGrowthRate: Math.max(0, preset.costGrowthRate + (Math.random() - 0.5) * 0.015),
      competitorPressureDrift: preset.competitorPressureDrift + (Math.random() - 0.5) * 0.01,
      riskDrift: preset.riskDrift + (Math.random() - 0.5) * 0.01,
    };

    let lastStep: StepResult | null = null;
    for (let period = 1; period <= preset.periods; period += 1) {
      const step = computeOneStep(state);
      lastStep = { period, ...step };
      state = evolveState(state, randomizedPreset, step);
    }

    if (lastStep) {
      success.push(lastStep.successScore);
      profit.push(lastStep.profit);
      customers.push(lastStep.customerCount);
      revenue.push(lastStep.revenue);
    }
  }

  return {
    avgFinalSuccessScore: mean(success),
    p10FinalSuccessScore: percentile(success, 10),
    p50FinalSuccessScore: percentile(success, 50),
    p90FinalSuccessScore: percentile(success, 90),
    avgFinalProfit: mean(profit),
    avgFinalCustomerCount: mean(customers),
    avgFinalRevenue: mean(revenue),
  };
}

function NumericInput({ label, value, onChange, step = 1 }: { label: string; value: number; onChange: (v: number) => void; step?: number }) {
```
